#include <stdio.h>
#include <stdbool.h>

// Function to check if a number is prime
bool isPrime(int num) {
    if (num <= 1) {
        return false;
    }
    for (int i = 2; i * i <= num; i++) {
        if (num % i == 0) {
            return false;
        }
    }
    return true;
}

// Function to print the sum of the first n prime numbers
void sumOfPrimes(int n) {
    int sum = 0;
    int count = 0;
    int num = 2; // Starting from the first prime number
    while (count < n) {
        if (isPrime(num)) {
            sum += num;
            count++;
        }
        num++;
    }
    printf("Sum of the first %d prime numbers is: %d\n", n, sum);
}

int main() {
    int n;
    printf("Enter the value of n: ");
    scanf("%d", &n);
    sumOfPrimes(n);
    return 0;
}
